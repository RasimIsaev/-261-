<div style="font-size:14px">
<p>Калькулятор работает по новым правилам <a href="https://dogovor-urist.ru/кодексы/гк_рф_1/ст_395/" target="_blank">статьи 395 ГК РФ</a>, принятой 1 июня 2015 г. и 1 августа 2016 г.</p>
<p>Однако калькулятор учитывает и ставки рефинансирования ЦБ, действовавшие <a href="https://dogovor-urist.ru/кодексы/гк_рф_1/ст_395/ред-01.01.1995/" target="_blank">до 1 июня 2015 года</a></p>
<p><b>О ставках в долларах и евро</b>: Учтены положения, освещаемые в Вопросе №3 раздела "Разъяснения по вопросам, возникающим в судебной практике" из "Обзора судебной практики Верховного Суда Российской Федерации N 1 (2017)" (утв. Президиумом Верховного Суда РФ 16.02.2017).
<br><a href="#faq1">Ознакомиться с подробностями</a>
</p>
<p>Калькулятор считает неустойку по договору поставки, подряда, аренды и другим видам неисполнения денежных обязательств в рублях, долларах или евро.</p>
<p>Если в самом договоре <b>прописан процент</b> неустойки, то расчёт нужно проводить с помощью <a target="_blank" href="/calculator/dolg_fixed_percent/">калькулятора расчёта договорной неустойки с фиксированным процентом</a></p>

function resAddData(resData, sum, index, dateStart, dateFinish, district, loanType) {

	if (loanType == DAYS_30) {
		resData.push({type: DATA_TYPE_INFO, data: processData(sum, index, dateStart, dateFinish, district, 360, loanType)});
		return;
	}
	var y1 = dateStart.getFullYear();
	var y2 = dateFinish.getFullYear();

	if (y1 == y2)
		resData.push({type: DATA_TYPE_INFO, data: processData(sum, index, dateStart, dateFinish, district, getDaysInYear(y1), loanType)});
	else {
		var start = y1;
		var d1 = getDaysInYear(y1);
		var i;
		for (var i = y1 + 1; i <= y2; i++) {
			var d2 = getDaysInYear(i);
			if (d1 != d2) {
				var ds = (start == y1)? dateStart : new Date(start, 0, 1);
				var df = new Date(i, 0, 0);
				resData.push({type: DATA_TYPE_INFO, data: processData(sum, index, ds, df, district, d1, loanType)});
				if (i == y2) {
					resData.push({type: DATA_TYPE_INFO, data: processData(sum, index, new Date(i, 0, 1), dateFinish, district, d2, loanType)});
				}
				start = i;
				d1 = d2;
			}
		}
		if (start != y2) {
			var ds = (start == y1)? dateStart : new Date(start, 0, 1);
			var df = dateFinish;
			resData.push({type: DATA_TYPE_INFO, data: processData(sum, index, ds, df, district, d1, loanType)});
		}
	}
}

function gluePayments(payments, loans) {
	var res = [];
	var i1 = 0, i2 = 0;
	while (i1 < payments.length || i2 < loans.length) {
		if (i1 >= payments.length) {
			var d = loans[i2];
			res.push({date: new Date(d.date.getTime() - ONE_DAY), sum : -d.sum, order : d.order, datePlus : d.date});
			i2++;
		} else if (i2 >= loans.length) {
			var d = payments[i1];
			res.push({date: d.date, sum : d.sum, order : d.order, datePlus : d.datePlus});
			i1++;
		} else {
			if (payments[i1].date.getTime() < loans[i2].date.getTime()) {
				var d = payments[i1];
				res.push({date: d.date, sum : d.sum, order : d.order, datePlus : d.datePlus});
				i1++;
			} else {
				var d = loans[i2];
				res.push({date: new Date(d.date.getTime() - ONE_DAY), sum : -d.sum, order : d.order, datePlus : d.date});
				i2++;
			}
		}
	}
	return res;
}

function cutLink(link) {
	var len = link.length;
	if (len < 50)
		return link;
	return link.substring(0, 40) + ' ... ' + link.substring(len - 10, len);
}

function preparePayments(payments) {
	if (!payments || !payments.length)
		return '';
	var res = '';
	for (var i = 0; i < payments.length; i++) {
		var p = payments[i];
		res += ';' + fd(p.date) + '_' + p.sum + '_' + p.order;
	}
	return res.substring(1);
}

function processData(sum, index, dateStart, dateFinish, district, daysInYear, loanType) {
	var days = dateDiff(dateFinish, dateStart, loanType);
	return {ind: index, cost: countCost(sum, days, index, district, daysInYear), days:days, dateStart:dateStart, dateFinish:dateFinish, sum:sum, daysInYear: daysInYear};
}

function collectPaymentsEx(prefix) {
	var form = document.forms['calcTable'];
	var res = [];
	var payDates = form.elements[prefix + 'Dates[]'];
	var val;
	if (payDates.length) {// больше, чем 2 оплаты
		var paySums = form.elements[prefix + 'Sums[]'];
		var payOrders = form.elements[prefix + 'Orders[]'];
		for (var i = 0; i < payDates.length; i++) {
			val = testPaymentLine(payDates[i], paySums[i], payOrders[i]);
			if (!val)
				return null;
			if (val.date != null)
				res.push(val);
		}
	} else {
		val = testPaymentLine(form.elements[prefix + 'Dates[]'], form.elements[prefix + 'Sums[]'], form.elements[prefix + 'Orders[]']);
		if (!val)
			return null;
		if (val.date != null)
			res.push(val);
	}

	return sortPayments(res);
}

function sortPayments(arr) {
	for (var i = 0; i + 1 < arr.length; i++)
		for (var j = i + 1; j < arr.length; j++)
			if (arr[i].date > arr[j].date) {
				var tmp = arr[i];
				arr[i] = arr[j];
				arr[j] = tmp;
			}
	return arr;
}

function testPaymentLine(payDate, paySum, payOrder) {
	if (!(payDate.value && paySum.value))
		return {date: null, sum: null, order: null};
	var resDate = null;
	if (payDate.value) {
		resDate = dateParse(payDate.value);
	}

	if (!resDate) {
		wrongData($(payDate).attr('id'));
		return null;
	}

	var resSum = null;
	if (paySum.value) {
		resSum = normalizeLoan(paySum.value);
	}

	if (!resSum) {
		wrongData($(paySum).attr('id'));
		return null;
	}

	return {date: resDate, datePlus: new Date(resDate.getTime() + ONE_DAY), sum: resSum, order: payOrder.value};
}

function fd(date) {
	var day = date.getDate();
	if (day < 10)

var ONE_DAY = 1000*60*60*24;

var DAYS_ABS = 1;
var DAYS_30 = 2;
var DAYS_30_SMALL = 3;

function dateDiff(date1, date2, type) {
	var date1_ms = date1.getTime();
	var date2_ms = date2.getTime();
	var difference_ms = date1_ms - date2_ms;
	var absDiff = Math.round(difference_ms/ONE_DAY);

	if (type == DAYS_ABS) {
		return absDiff + 1;
	}

	var day1 = date1.getDate();
	var day2 = date2.getDate();
	var month2 = date2.getMonth() + 1;
	var month1 = date1.getMonth() + 1;
	var year2 = date2.getFullYear();
	var year1 = date1.getFullYear();

	if (day1 == daysInMonth(year1, month1))
		day1 = 30;
	return 30*(12*(year1-year2)+(month1-month2))+day1-day2+1;
}

function daysInMonth(year, month) {
	return new Date(year, month, 0).getDate();
}

function dateParse(dateStr) {
	if (dateStr == null) return null;
	var dateParts = dateStr.split(".");
	if (dateParts.length != 3)
		return null;
	var d = new Date(dateParts[2], (dateParts[1] - 1), dateParts[0]);
	if (d && !isNaN(d.getTime())) {
		var fy = parseInt(dateParts[2]);
		if (fy < 50)
			d.setFullYear(2000 + fy);
		else if (fy < 100)
			d.setFullYear(1900 + fy);
		return d;
	}
	return null;
}

function moneyFormat(money) {
	//money = normalizeLoan(money);
	return addCommas(parseFloat(money).toFixed(2)).replace(/\./g, ',');
}

function addCommas(nStr)
{
	nStr += '';
	x = nStr.split('.');
	x1 = x[0];
	x2 = x.length > 1 ? '.' + x[1] : '';
	var rgx = /(\d+)(\d{3})/;
	while (rgx.test(x1)) {
		x1 = x1.replace(rgx, '$1' + ' ' + '$2');
	}
	return x1 + x2;
}

function parseTableText(text) {
	var lines = text.split('\n');
	var res = [];
	for (var i = 0; i < lines.length; i++) {
		var data = lines[i].split('\t');
		if (data.length < 2) continue;
		var sum = normalizeLoan(data[1]);
		var date = dateParse(data[0]);
		if (!(date && sum))
			continue;

		res.push({date: date, sum: sum, order: (data.length > 2)? data[3] : ''});
	}
	return res;
}

<h3>
		Описание калькулятора расчёта процентов</h3>
	<p>
		При просрочке выплат уплате подлежат проценты на сумму задолженности. Размер процентов определяется в зависимости от места жительства физического лица-кредитора или места нахождения юридического лица-кредитора, опубликованным Банком России и имевшими место в соответствующие периоды по средним ставкам банковского процента по вкладам для физических лиц.</p>
	<p>
		Стоит отметить, что&nbsp;<strong>с 1 июня 2015 года</strong>&nbsp;расчёт для&nbsp;<strong>каждого федерального округа</strong>&nbsp;стал дифференцированным и каждый месяц рассчитывается по-новому</p>
	<p>
		Ссылка на сайт ЦБ&nbsp;<a href="//cbr.ru/statistics/?Prtid=int_rat&amp;ch=PAR_222#CheckedItem">Сведения о средних ставках банковского процента по вкладам физических лиц в рублях для целей применения ст. 395 ГК РФ</a>.</p>
	<p>
		Ранее 1 июня расчёт совершался одинаково для всех регионов РФ на основании ставки рефинансирования ЦБ (<a href="//cbr.ru/pw.aspx?file=/statistics/credit_statistics/refinancing_rates.htm">Ставка рефинансирования ЦБ РФ до 1 июня 2015 года</a>).&nbsp;</p>
	<p>
		Данное правило применятся в&nbsp; том случае, если размер процентов&nbsp;<strong>не установлен договором</strong>&nbsp;или законом. Если убытки, причиненные кредитору, превышают сумму процентов, он имеет право также требовать от должника возмещения убытков, которые превышают эту сумму.</p>
	<p>
		Чтобы быстро рассчитать задолженность вам достаточно ввести сумму задолженности, период просрочки и регион проживания или нахождения должника.</p>

	<p>

	<div class="comment-quote">
		<p>
			2. При расчете подлежащих уплате годовых процентов по ставке рефинансирования Центрального банка Российской Федерации число дней в году (месяце) принимается равным соответственно 360 и 30 дням, если иное не установлено соглашением сторон, обязательными для сторон правилами, а также обычаями делового оборота.</p>
		<p>
			Проценты начисляются до момента фактического исполнения денежного обязательства, определяемого исходя из условий о порядке платежей, форме расчетов и положений статьи 316 ГК РФ о месте исполнения денежного обязательства, если иное не установлено законом либо соглашением сторон.</p>

		<p>При расчёте процентов за пользование чужими денежными средствами необходимо правильно рассчитать период такой
	просрочки. Если срок платежа установлен конкретной датой (например, до 4 мая 2024 г., до 10 числа каждого месяца, и
	пр.), то период просрочки исчисляется со следующего дня (например, соответственно с 5 мая 2024 г., с 11 числа
	каждого месяца).</p>
<h3>Расчёт неустойки по периодическим платежам</h3>
<p>Важно отметить, что при расчёте процентов для периодических платежей (например, арендная плата) размер процентов за
	пользование чужими денежными средствами будет суммой всех периодов.</p>
<p>
	Пример. Заключен договор аренды квартиры сроком на 6 месяцев с января 2024 года. Размер арендной платы 20&nbsp;000 в
	месяц, оплата должна производиться до 10 числа каждого месяца. Если арендатор заплатил за первый месяц, а за
	остальные не заплатил, то проценты будут считаться за следующие периоды:</p>
<p style="margin-left:36pt;">
	1). 20&nbsp;000 руб., период просрочки с 11.02.2024 г. по 30.09.2024 (например, день подачи иска).</p>
<p style="margin-left:36pt;">
	2). 20&nbsp;000 руб., период просрочки с 11.03.2024 г. по 30.09.2024.</p>
<p style="margin-left:36pt;">
	3). 20&nbsp;000 руб., период просрочки с 11.04.2024 г. по 30.09.2024.</p>
<p style="margin-left:36pt;">
	4). 20&nbsp;000 руб., период просрочки с 11.05.2024 г. по 30.09.2024.</p>
<p style="margin-left:36pt;">
	5). 20&nbsp;000 руб., период просрочки с 11.06.2024 г. по 30.09.2024.</p>
<p>
	В итоге по каждому платежу будет своя сумма процентов за пользование чужими денежными средствами, а если все эти
	суммы сложить, то получится окончательный размер процентов за пользование чужими денежными средствами.</p>
<h3>Расчёт процентов без указанного срока возврата</h3>
<p>
	Иногда в жизни возникают ситуации, когда <b>не указано число или срок</b> исполнения денежного обязательства, то есть срок
	возврата денег не определён. В этом случае следует руководствоваться законом. Например, если в расписке указано
	только, что один человек получил от другого деньги в долг в сумме 100&nbsp;000 рублей. В данном случае срок возврата
	денег не указан, значит, необходимо применять закон (ч. 1 <a
			href="https://dogovor-urist.ru/кодексы/гражданский_кодекс/статья_810/">ст. 810 ГК РФ</a>):
	</p>
<div class="comment-quote">В случаях, когда срок возврата договором не установлен или определён моментом востребования, сумма
		займа должна быть возвращена заёмщиком <b>в течение тридцати дней</b> со дня предъявления займодавцем требования об
		этом, если иное не предусмотрено договором.</div>
<p>
	То есть для того, чтобы посчитать период просрочки, сначала <b>необходимо направить должнику</b> требование о возврате
	денег и отсчитать 30 дней с момента получения этого требования.</p>
<p>Период просрочки начинается на 31 день после 	получения требования - это будет первый день просрочки.
	<br><em>Например, должник получил требование о возврате денег 10
	июля. Значит, период просрочки будет считаться так: 10 июля + 30 дней на исполнение. 10 августа &ndash; последний
	день возврата долга. Начало периода просрочки с 11 августа.</em></p>
<h3>Неустойка по самостоятельно установленным процентам (Договорная неустойка)</h3>
<p>
	Действующее законодательство допускает <b>самостоятельное установление процентов</b> сторонами, не ограничивая размер таких
	процентов, то есть, если стороны согласны, что начисляется 1% в день за каждый день просрочки от суммы, которая не
	возращена, то именно такой размер и учитывается.</p>
<p><b>НО несмотря на это</b>, суд может по заявлению должника уменьшить этот
	процент, если он несоразмерен последствиям невыплаты долга <a
			href="https://dogovor-urist.ru/кодексы/гражданский_кодекс/статья_395/">(ч. 6 ст. 395 ГК РФ</a>), но не меньше
	чем до средней ставки банковского процента по вкладам физических лиц.</p>
<div class="comment-quote">
	<p>
		6. Если подлежащая уплате сумма процентов явно несоразмерна последствиям нарушения обязательства, суд по
		заявлению должника вправе уменьшить предусмотренные договором проценты, но не менее чем до суммы, определенной
		исходя из ставки, указанной в&nbsp;пункте&nbsp;1&nbsp;настоящей статьи.<br/>
		<br/>
		В связи с вышеизложенным иногда нецелесообразно заявлять завышенные исковые требования, например, если сумма
		долга 100&nbsp;000 рублей, а проценты за пользование этой суммой 200&nbsp;000 рублей, то можно сразу уменьшить
		размер процентов, не дожидаясь, когда другая сторона попросит об этом суд. Ведь именно с суммы иска Вы
		оплачиваете государственную пошлину, так зачем переплачивать, зная, что размер процентов явно будет уменьшен.
	</p>
</div>
<h3>Примечание по калькулятору</h3>
<p>
	Важно понимать, что для разных ситуаций применяются разные нормы закона. Например, для расчёта процентов за задержку
	выплаты заработной платы применяется свой размер процентов (<a target="_blank" href="/калькуляторы/калькулятор_расчёта_процентов_за_задержку_зарплаты/">см. калькулятор по расчёту процентов задолженности по зарплате</a>),
	так как в данном случае действует не нормы гражданского, а нормы трудового права.</p>
<p>
	Также существует и другие случаи, когда размер процента чётко установлен, но в другом размере, нежели <a
			href="https://dogovor-urist.ru/кодексы/гражданский_кодекс/статья_395/">ст. 395 ГК РФ</a>. <em><a
				href="https://dogovor-urist.ru/законы/о_защите_прав_потребителей/статья_23/">Ст. 23</a> &laquo;Закона о
		защите прав потребителей&raquo;</em>&nbsp;гласит,&nbsp;</p>
<div class="comment-quote">
	<p>
		За нарушение предусмотренных&nbsp;статьями 20,&nbsp;21&nbsp;и&nbsp;22&nbsp;настоящего Закона сроков, а также за
		невыполнение (задержку выполнения) требования потребителя о предоставлении ему на период ремонта (замены)
		аналогичного товара продавец (изготовитель, уполномоченная организация или уполномоченный индивидуальный
		предприниматель, импортер), допустивший такие нарушения, <strong>уплачивает потребителю за каждый день просрочки&nbsp;неустойку</strong><strong>
			&nbsp;(пеню) <u>в размере одного процента цены товара</u></strong><u>.</u></p>

	<p>
		Цена товара определяется, исходя из его цены, существовавшей в том месте, в котором требование потребителя
		должно было быть удовлетворено продавцом (изготовителем, уполномоченной организацией или уполномоченным
		индивидуальным предпринимателем, импортером), в день добровольного удовлетворения такого требования или в день
		вынесения судебного решения, если требование добровольно удовлетворено не было.</p>
</div>
<h3>
	Вывод по калькулятору</h3>
<p>
	Необходимо понимать, что для гражданского права (всевозможные договора, например, займа, купли-продажи, аренды,
	поставки, оказания услуг и пр.) применяются нормы Гражданского кодекса РФ (<a
			href="https://dogovor-urist.ru/кодексы/гражданский_кодекс/статья_395/">ст. 395 ГК РФ</a> или договорная
	неустойка, размер которой указывается в самом договоре), для трудовых отношений &ndash; нормы Трудового кодекса РФ (<a
			href="https://dogovor-urist.ru/кодексы/трудовой_кодекс/статья_236/">ст. 236 ТК РФ</a>). Если же дело касается
	защиты прав потребителей, то применяется закон &laquo;О защите прав потребителей&raquo;.</p>
<p>
	<strong>Если же Вы сомневаетесь в том или ином вопросе, Вы всегда можете обратиться за консультацией к юристам
		нашего сайта.</strong></p>	</div>
	<div id="calc-panos2" style="display: none">
		<p>
	Если Вы кому-то дали в долг, оформив это распиской или договором займа, а долг все не возвращают, то можно для защиты своих прав обратиться в суд.</p>
<h3>
	1. Место подачи иска</h3>
<p>
	По общему правилу иск подаётся по месту жительства ответчика (<a href="https://dogovor-urist.ru/кодексы/гпк_рф/ст_28/">ст. 28 ГПК РФ</a>).</p>
<p>
	Есть и другие варианты (например, если ответчик юридическое лицо, то иск может быть предъявлен по месту нахождения филиала <a href="https://dogovor-urist.ru/кодексы/гпк_рф/ст_29/">ст. 29 ГПК РФ</a>).</p>
<h3>
	2. В какой суд обращаться?</h3>
<p>
	Необходимо различать подсудность, в какой суд подавать исковое заявление: <strong>мировому судье</strong> или в <strong>районный суд</strong>.</p>
<p>Если долг меньше 50&nbsp;000 рублей и есть договор займа, расписка, то нужно обращаться к <strong>мировому судье</strong> (ст. <a href="https://dogovor-urist.ru/кодексы/гпк_рф/ст_23/">23</a>, <a href="https://dogovor-urist.ru/кодексы/гпк_рф/ст_24/">24</a> ГПК РФ). Таким образом, подсудность зависит от категории дела, суммы иска и исковых требований.</p>
<p>В случае, если исковые требования более 50 000 рублей или есть требование о взыскании морального вреда, то иск подаётся в районный суд.</p>
<h3>
	3. Составление искового заявления.</h3>
<p>
	Для обращения в суд требуется составить исковое заявление. В нём описываются подробно все обстоятельства произошедшего, кроме того, указывается точная сумма, которую Вы просите взыскать.</p>
<p>Например, если Вы дали в долг 40&nbsp;000 рублей, но 5000 руб. Вам уже вернули, то исковые требования будут о возврате 35&nbsp;000 рублей и процентов за пользование чужими денежными средствами и пр.</p>
<p>В сумму иска в данном случае не включаются суммы по судебным расходам (госпошлина, оказание юридической помощи). То есть, если заявлено требование о взыскании госпошлины с ответчика, то эта сумма в цену иска не засчитывается.
<p>Например, если истец просит взыскать сумму долга 35 000 рублей, 9 000 рублей проценты за пользование чужими денежными средствами и судебные расходы 10 000 рублей, то сумма иска будет 44 000 рублей. Соответственно, иск подаётся мировому судье.</p>
<h3>4. Документ расчёта процентов</h3>
<p>
	Помимо договора, расписки и др. документов обязательно <strong>предоставить расчёт процентов</strong>.</p>
<p>
	Данный расчёт можно включить как в текст самого искового заявления, так и выполнить в качестве отдельного документа, который соответственно и назвать: &laquo;Расчёт процентов&raquo;.</p>
<p>
	При расчёте процентов важно учесть, что процент за пользование чужими денежными средствами может быть <strong>установлен договором</strong>.</p>
<p>
	Если в самом договоре или расписке нет такого условия, то применяется процент, указанный в законе (<a href="https://dogovor-urist.ru/кодексы/гк_рф_1/ст_395/">ст. 395 ГК РФ</a>), и его можно расчитать калькулятором, на странице которого Вы находитесь.</p>
<p>
	Если в договоре или расписке прописан конкретный процент, то применяется именно он.</p>
<h3>
	5. Госпошлина</h3>
<p>
	При подаче иска уплачивается госпошлина. Уплачивается она до подачи иска, так как квитанция об оплате госпошлины предоставляется вместе с исковым заявлением в суд &ndash; <strong>обязательно оригинал квитанции</strong>.</p>
<p>
	Расчёт госпошлины производится в соответствии с налоговым кодексом Российской Федерации и напрямую зависит от цены иска, а также от того, в каком суде рассматривается дело (мировой судья, районный суд, арбитражный суд).</p>
<h3>
	6. Полный перечень документов для подачи иска</h3>
<p>
	Перечень документов, которые прилагаются к иску, установлен <a href="https://dogovor-urist.ru/кодексы/гпк_рф/ст_132/">ст. 132 ГПК РФ</a></p>
<div class="comment-quote">
	<p>
		Статья 132. Документы, прилагаемые к исковому заявлению</p>
	<p>
				К исковому заявлению прилагаются:</p>
	<ul>

		<li>
			его копии в соответствии с количеством ответчиков и третьих лиц;</li>
		<li>

<h3>
	7. Обращение в суд по взысканию задолженности</h3>
<p>
	Когда исковое заявление готово и все документы собраны, можно обращаться в суд. Исковое заявление можно подать непосредственно в суд, при этом Вам поставят отметку о принятии иска на Вашем экземпляре иска. Либо можно направить заказным письмом с описью вложений.</p>
<p>
	После этого Вам останется только ждать, когда суд известит Вас о дате судебного заседания.</p>

<h2>Часто задаваемые вопросы</h2>
<a name="faq1"></a><h3>1. Откуда брать ставки для расчётов по ст. 395 ГК для долларов и евро?</h3>

<p>В Вопросе №3 раздела "Разъяснения по вопросам, возникающим в судебной практике" из "Обзора судебной практики Верховного Суда Российской Федерации N 1 (2017)" (утв. Президиумом Верховного Суда РФ 16.02.2017) освещается данная проблема:</p>
<div class="article-notice">
<p><b>ВОПРОС 3</b>. На основании какой ставки определяется размер процентов, уплачиваемых за неправомерное удержание денежных средств, уклонения от их возврата, иной просрочки в их уплате, в случае, когда в соответствии с законодательством о валютном регулировании и валютном контроле при осуществлении расчетов по обязательствам допускается использование иностранной валюты и денежное обязательство выражено в ней, а равно когда денежное обязательство подлежит оплате в рублях в сумме, эквивалентной определенной сумме в иностранной валюте или в условных денежных единицах?</p>
<p><b>ОТВЕТ</b>. В соответствии с п. 1 ст. 395 ГК РФ в случаях неправомерного удержания денежных средств, уклонения от их возврата, иной просрочки в их уплате подлежат уплате проценты на сумму долга. Размер процентов определяется ключевой ставкой Банка России, действовавшей в соответствующие периоды. Эти правила применяются, если иной размер процентов не установлен законом или договором.</p>
<p>Указанный порядок расчета процентов подлежит применению в случаях, когда денежное обязательство выражено в рублях, поскольку ключевая ставка Банка России, являясь основным индикатором денежно-кредитной политики, устанавливается Центральным Банком Российской Федерации в процентном отношении и используется при расчетах в национальной валюте - рубль.</p>
<p>Вместе с тем в соответствии с законодательством о валютном регулировании и валютном контроле в определенных случаях денежное обязательство может быть выражено в иностранной валюте, может допускаться использование иностранной валюты при осуществлении расчетов по обязательствам. Также в самом денежном обязательстве может быть предусмотрено, что оно подлежит оплате в рублях в сумме, эквивалентной определенной сумме в иностранной валюте или в условных денежных единицах (пп. 1, 2 ст. 317 ГК РФ ).</p>
<p>Исходя из толкования ст. 309, 317 и 395 ГК РФ в их взаимосвязи при просрочке исполнения денежного обязательства, валютой долга которого является иностранная валюта, проценты за неправомерное удержание денежных средств, уклонения от их возврата, иной просрочки в их уплате подлежат исчислению в иностранной валюте, поскольку целью уплаты указанных процентов является восстановление имущественного положения кредитора и компенсация неполученного им дохода от возможного использования денежных средств, не возвращенных в срок должником.</p>
<p>Принимая во внимание, что ключевая ставка Банка России представляет собой процентную ставку по операциям предоставления Банком России коммерческим банкам краткосрочных кредитов на аукционной основе, размер процентов, уплачиваемых за нарушение денежного обязательства, валютой долга которого является иностранная валюта, должен определяться с учетом аналогичных показателей и исчисляться исходя из средних процентных ставок в валюте долга.</p>
<p><b>Источниками информации о средних ставках по краткосрочным кредитам в иностранной валюте являются официальный сайт Банка России в сети "Интернет" и официальное издание Банка России "Вестник Банка России"</b>.</p>
<p>Если средняя ставка в соответствующей иностранной валюте за определенный период не опубликована, размер подлежащих взысканию процентов устанавливается исходя из самой поздней из опубликованных ставок по каждому из периодов просрочки.</p>
<p>Когда отсутствуют и такие публикации, сумма подлежащих взысканию процентов рассчитывается на основании справки одного из ведущих банков в месте нахождения кредитора, <b>подтверждающей применяемую им среднюю ставку по краткосрочным валютным кредитам</b></p>
</div>
<p>Сведения о ставках мы берём с <a target="_blank" href="https://www.cbr.ru/statistics/?PrtId=int_rat">данной страницы Центального Банка РФ</a>
из раздела "Сведения по кредитам в рублях, долларах США и евро | в целом по Российской Федерации".
</p>

<a name="faq2"></a><h3>2. Почему калькулятор берёт 365/366 дней в году, а не 360?</h3>
<p>Калькулятор умеет считать обоими способами. Однако мы категорически рекомендуем использовать пункт <b>в календарных днях</b></p>
<p>Действительно, Постановления Пленума Верховного Суда РФ N 13, Пленума ВАС РФ N 14 от 08.10.1998 (далее - Постановление N 13/14) было указано:</p>
<div class="article-notice">При расчете подлежащих уплате годовых процентов по ставке рефинансирования Центрального банка Российской Федерации число дней в году (месяце) принимается равным соответственно 360 и 30 дням, если иное не установлено соглашением сторон, обязательными для сторон правилами, а также обычаями делового оборота.</div>
<p>Однако этот <b>пункт был отменён</b> Пунктом 84 Постановления Пленума Верховного Суда РФ от 24.03.2016 № 7</p>
